# js干货、小技巧

## 获取网络图片宽高
```js
function getImgWH(imgUrl) {
    const image = new Image();
    image.src = imgUrl;
    return new Promise(function(resolve) {
        if (image.complete) {
            // 图片被缓存了
            resolve({
                width: image.width,
                heigh: image.height
            })
        } else {
            image.onload = function() {
                resolve({
                    width: image.width,
                    heigh: image.height
                })
            }
        }
    })
}
```
